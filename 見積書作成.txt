const loadStorage = (key, fallback) => {
  try {
    const raw = localStorage.getItem(key);
    return raw ? JSON.parse(raw) : fallback;
  } catch {
    return fallback;
  }
};

// ★ パフォーマンス最適化: 同一値書込みをスキップ（localStorage は同期I/Oで重い）
const saveStorageCache = {};
const saveStorage = (key, data) => {
  try {
    const serialized = JSON.stringify(data);
    if (saveStorageCache[key] === serialized) return true; // 同一なら何もしない
    saveStorageCache[key] = serialized;
    localStorage.setItem(key, serialized);
    return true;
  } catch {
    return false;
  }
};

// ===== 商品名からの推測ロジック =====
const inferFromProductName = (name) => {
  if (!name) return null;
  const n = name.toLowerCase();
  let volume = null, unit = null, baseScale = null;

  const match = n.match(/(\d+\.?\d*)\s*(ml|l|g|kg|枚|個|本|箱|m|cm|組|台)/i);
  if (match) {
    volume = match[1];
    const foundUnit = match[2].toLowerCase();
    const option = UNIT_OPTIONS.find(
      (o) => o.value.toLowerCase() === foundUnit || (foundUnit === 'l' && o.value === 'L')
    );
    if (option) {
      unit = option.value;
      baseScale = option.defaultBase;
    }
  }

  if (!unit) {
    if (n.includes('トイレット') || n.includes('ロール')) unit = '個';
    else if (n.includes('アルコール') || n.includes('石鹸') || n.includes('液') || n.includes('洗剤')) unit = 'ml';
    else if (n.includes('袋') || n.includes('グローブ') || n.includes('マスク')) unit = '枚';
    else if (n.includes('ラップ')) unit = 'm';
    else if (n.includes('車椅子') || n.includes('車いす') || n.includes('ベッド')) unit = '台';

    if (unit) {
      const option = UNIT_OPTIONS.find((o) => o.value === unit);
      baseScale = option?.defaultBase || 1;
    }
  }
  return { volume, unit, baseScale };
};

  if (raw === null || raw === undefined) return '';
  const s = String(raw)
    .replace(/[０-９]/g, (ch) => String.fromCharCode(ch.charCodeAt(0) - 0xFEE0))
    .replace(/[,，\s]/g, '');
  return s;

        lastDraftStringRef.current = serialized;
        localStorage.setItem(STORAGE_KEYS.DRAFT, serialized);
      } catch (err) {
        console.warn('Draft save error:', err);
      }

      if (draftSaveTimerRef.current) clearTimeout(draftSaveTimerRef.current);
    };

      validityPeriod, paymentTerms, deliveryInfo, remarks, bottomRemarks,

  const calculatePrices = (item, changedField) => {
    let { packSize, price, casePrice, cost, caseCost, targetMargin } = item;
    const ps = Math.max(Number(packSize) || 1, 1);

    // 利益率の自動算出ヘルパー：原価と価格の両方が入力されていれば利益率を逆算
    const recalcMargin = () => {
      const p = Number(price) || 0;
      const c = Number(cost) || 0;
      if (p > 0 && c > 0 && c < p) {
        // 利益率（粗利率）= (販売価格 - 原価) / 販売価格 × 100
        targetMargin = Math.round(((p - c) / p) * 1000) / 10; // 小数1桁
      } else if (p > 0 && c === 0) {
        targetMargin = '';
      }
    };

    if (changedField === 'packSize') {
      casePrice = (Number(price) || 0) * ps;
      caseCost = (Number(cost) || 0) * ps;
    } else if (changedField === 'casePrice') {
      price = Math.round((Number(casePrice) || 0) / ps);
      recalcMargin(); // 価格変更時：原価があれば利益率を逆算
    } else if (changedField === 'caseCost') {
      cost = Math.round((Number(caseCost) || 0) / ps);
      // 原価変更時：すでに目標利益率があれば価格を再計算、なければ価格から逆算
      if (targetMargin !== '' && Number(targetMargin) < 100 && Number(targetMargin) > 0 && (Number(price) || 0) === 0) {
        price = Math.round(cost / (1 - Number(targetMargin) / 100));
        casePrice = price * ps;
      } else {
        recalcMargin();
      }
    } else if (changedField === 'price') {
      casePrice = (Number(price) || 0) * ps;
      recalcMargin(); // 販売価格変更時：原価があれば利益率を逆算表示
    } else if (changedField === 'cost') {
      caseCost = (Number(cost) || 0) * ps;
      if (targetMargin !== '' && Number(targetMargin) < 100 && Number(targetMargin) > 0 && (Number(price) || 0) === 0) {
        price = Math.round(cost / (1 - Number(targetMargin) / 100));
        casePrice = price * ps;
      } else {
        recalcMargin();
      }
    } else if (changedField === 'targetMargin') {
      // 利益率を手入力した場合：原価から販売価格を逆算
      if (targetMargin !== '' && Number(targetMargin) < 100 && Number(targetMargin) > 0) {
        const c = Number(cost) || 0;
        if (c > 0) {
          price = Math.round(c / (1 - Number(targetMargin) / 100));
          casePrice = price * ps;
        }
      }
    }
    return { ...item, packSize, price, casePrice, cost, caseCost, targetMargin };
  };

  const computedGroups = useMemo(() => {
    return groups.map((group) => {
      const monthlyUsage = (group.baseVolume || 0) * (group.baseQuantity || 0);
      const cleanTitle = group.title ? group.title.replace(CLEAN_TITLE_REGEX, '').trim() : '';

      const calculatedItems = group.items.map((item) => {
        const vol = Number(item.volume);
        const pri = Number(item.price);
        const cost = Number(item.cost) || 0;
        const profit = pri - cost;
        const margin = pri > 0 ? (profit / pri) * 100 : 0;
        const unitPrice = vol > 0 ? (pri / vol) * group.baseScale : 0;
        const monthlyCost = vol > 0 ? (monthlyUsage / vol) * pri : 0;
        return {
          ...item,
          profit,
          margin: Math.round(margin * 10) / 10,
          unitPrice: Math.round(unitPrice * 10) / 10,
          unitPriceRaw: unitPrice,
          monthlyCost: Math.round(monthlyCost),
          yearlyCost: Math.round(monthlyCost * 12),
        };
      });

      if (calculatedItems.length === 0) return { ...group, cleanTitle, items: [], stats: null };

      const sortedForStats = [...calculatedItems].sort((a, b) => b.unitPriceRaw - a.unitPriceRaw);
      const best = sortedForStats[sortedForStats.length - 1];
      const worst = sortedForStats[0];
      const discountRate = worst.unitPriceRaw > 0
        ? Math.round(((worst.unitPriceRaw - best.unitPriceRaw) / worst.unitPriceRaw) * 100) : 0;
      const yearlyReduction = worst.yearlyCost - best.yearlyCost;

      return {
        ...group, cleanTitle, items: calculatedItems,
        stats: { minItem: best, maxItem: worst, discountRate, yearlyReduction, monthlyUsage }
      };
    });

  const parseCSVLine = (line) => {
    const out = [];
    let cur = '';
    let inQuotes = false;
    for (let i = 0; i < line.length; i++) {
      const ch = line[i];
      if (inQuotes) {
        if (ch === '"' && line[i+1] === '"') { cur += '"'; i++; }
        else if (ch === '"') { inQuotes = false; }
        else { cur += ch; }
      } else {
        if (ch === '"') inQuotes = true;
        else if (ch === ',') { out.push(cur); cur = ''; }
        else cur += ch;
      }
    }
    out.push(cur);
    return out;

                          })()}
                        </td>
                        <td colSpan={(showCost ? 2 : 0) + (showSimulation ? 2 : 1) + 1}
                          className="px-3 py-2 text-right align-top pt-3">
                          <button onClick={() => handleAddItem(group.id)} disabled={!canAdd}
                            className={`px-3 py-1.5 rounded text-xs font-bold transition w-full ${canAdd ? 'bg-slate-800 text-white hover:bg-slate-700' : 'bg-slate-200 text-slate-400 cursor-not-allowed'}`}>
                            <Plus className="w-3 h-3 inline mr-0.5" /> 追加
                          </button>
                        </td>
                      </tr>
                    )}
                  </tbody>
                </table>
              </div>

              {/* === 手動ページ切替（前へ／次へ）=== */}
              {!isPreviewMode && group.items.length > ITEMS_PER_PAGE && (
                <div className="flex items-center justify-center gap-3 px-4 py-2 bg-slate-50 border-t border-slate-200 no-print">
                  <button onClick={() => setCurrentPage(p => Math.max(1, p - 1))}
                    disabled={currentPage <= 1}
                    className={`px-4 py-1.5 rounded text-sm font-bold transition ${currentPage <= 1 ? 'bg-slate-200 text-slate-400 cursor-not-allowed' : 'bg-white text-slate-700 border border-slate-300 hover:bg-slate-100'}`}>
                    ← 前へ
                  </button>
                  <span className="text-sm font-bold text-slate-600">
                    {currentPage} / {Math.ceil(group.items.length / ITEMS_PER_PAGE)} ページ
                  </span>
                  <button onClick={() => setCurrentPage(p => Math.min(Math.ceil(group.items.length / ITEMS_PER_PAGE), p + 1))}
                    disabled={currentPage >= Math.ceil(group.items.length / ITEMS_PER_PAGE)}
                    className={`px-4 py-1.5 rounded text-sm font-bold transition ${currentPage >= Math.ceil(group.items.length / ITEMS_PER_PAGE) ? 'bg-slate-200 text-slate-400 cursor-not-allowed' : 'bg-white text-slate-700 border border-slate-300 hover:bg-slate-100'}`}>
                    次へ →
                  </button>
                </div>
              )}

              {/* 試算情報 */}
              {showSimulation && (
                <div className="bg-blue-50/50 px-4 py-2.5 border-t border-slate-200 flex flex-col md:flex-row justify-between items-center gap-3 print:bg-slate-50 print:py-1.5">
                  <div className="flex items-center gap-2 text-xs">
                    <span className="font-bold text-slate-600">試算条件:</span>
                    <div className="flex items-center bg-white px-2 py-1 rounded border border-slate-200">
                      <Calculator className="w-3 h-3 text-slate-400 mr-1" />
                      {isPreviewMode ? <span className="font-bold text-slate-700">{group.baseVolume}</span> :
                        <input type="number" value={group.baseVolume}
                          onChange={(e) => updateGroupField(group.id, 'baseVolume', e.target.value)}
                          className="w-14 text-right outline-none font-bold text-slate-700 bg-transparent" />}
                      <span className="text-[10px] text-slate-500 mx-1">{group.unit}</span>
                      <MultiplyIcon className="w-3 h-3 text-slate-400" />
                      {isPreviewMode ? <span className="font-bold text-slate-700 ml-1">{group.baseQuantity}</span> :
                        <input type="number" value={group.baseQuantity}
                          onChange={(e) => updateGroupField(group.id, 'baseQuantity', e.target.value)}
                          className="w-12 text-right outline-none font-bold text-slate-700 ml-1 bg-transparent" />}
                      <span className="text-[10px] text-slate-500 ml-1">個/月</span>
                    </div>
                  </div>
                  <div className="flex items-center gap-3 text-xs">
                    {group.stats?.discountRate > 0 && (
                      <div className="flex items-center text-red-700 bg-red-50 px-2 py-1 rounded border border-red-200 font-bold">
                        <Percent className="w-3 h-3 mr-1" />約 {group.stats.discountRate}% ダウン
                      </div>
                    )}
                    <div>
                      <span className="text-slate-500">年間削減見込：</span>
                      <span className={`font-bold text-base ml-1 ${group.stats?.yearlyReduction > 0 ? 'text-red-600' : 'text-slate-400'}`}>
                        ¥{(group.stats?.yearlyReduction || 0).toLocaleString()}
                      </span>
                    </div>
                  </div>
                </div>
              )}
            </div>
          );
        })}

        {/* カテゴリ追加ボタン */}
        {!isPreviewMode && (
          <button onClick={addGroup}
            className="w-full py-4 border-2 border-dashed border-slate-300 rounded-lg text-slate-400 hover:text-slate-700 hover:border-slate-500 hover:bg-slate-50 transition flex items-center justify-center gap-2 no-print">
            <Plus className="w-5 h-5" />
            <span className="font-bold">新しい商品カテゴリを追加</span>
          </button>
        )}

        {/* === 全商品下部の備考欄（新規） === */}
        <div className="bg-white rounded-lg shadow-sm border border-blue-200 p-3 mt-4">
          <label className="text-xs font-bold text-blue-700 mb-1 block">備考（見積書下部に印字されます）</label>
          {isPreviewMode ? (
            <div className="text-sm text-slate-800 whitespace-pre-wrap min-h-[1.5em]">{bottomRemarks || '—'}</div>
          ) : (
            <textarea value={bottomRemarks} onChange={(e) => setBottomRemarks(e.target.value)}
              placeholder="本見積書の記載金額には、消費税は含まれておりません。"
              rows={2}
              className="w-full bg-slate-50 border border-slate-200 rounded px-3 py-1.5 text-sm outline-none focus:bg-white focus:border-blue-500 resize-none" />
          )}
        </div>

        {/* 下部余白 */}
        <div className="pb-20 print:pb-0"></div>
      </main>

      {/* === 年間削減サマリー（固定フッター） === */}
      {showSimulation && totalReduction > 0 && (
        <div className="bg-white border-t border-slate-200 shadow-lg sticky bottom-0 z-40 p-3 no-print">
          <div className="max-w-[1100px] mx-auto flex justify-end items-center gap-3">
            <span className="text-xs font-bold text-slate-600">年間削減総額（試算）:</span>
            <span className="text-xl font-bold text-red-600">¥{totalReduction.toLocaleString()}</span>
          </div>
        </div>
      )}

      {/* === 通知 === */}
      {notification && (
        <div className="fixed bottom-4 right-4 bg-slate-900 text-white px-4 py-2.5 rounded-lg shadow-xl z-[100] text-sm font-bold animate-fade-in no-print">
          {notification}
        </div>
      )}

      {/* ===== 各モーダル ===== */}

      {/* 営業所設定モーダル */}
      {showOfficeModal && (
        <div className="fixed inset-0 bg-slate-900/60 z-[80] flex items-center justify-center backdrop-blur-sm p-4 no-print">
          <div className="bg-white rounded-xl shadow-2xl max-w-lg w-full max-h-[90vh] flex flex-col overflow-hidden">
            <div className="flex justify-between items-center p-4 border-b bg-amber-50 shrink-0">
              <h3 className="text-lg font-bold text-amber-800 flex items-center gap-2">
                <Building2 className="w-5 h-5" /> 営業所・会社情報の設定
              </h3>
              <button onClick={() => setShowOfficeModal(false)} className="text-slate-400 hover:text-slate-600">
                <X className="w-5 h-5" />
              </button>
            </div>
            <div className="flex-1 overflow-y-auto p-5 space-y-3">
              {/* 営業所マスタから選択 */}
              <div className="bg-amber-50 border border-amber-300 rounded-lg p-3 mb-2">
                <label className="text-xs font-bold text-amber-800 block mb-1.5 flex items-center gap-1">
                  <Building2 className="w-3.5 h-3.5" /> 営業所マスタから選択（自動入力）
                </label>
                <select
                  onChange={(e) => {
                    const selected = OFFICE_MASTER.find((o) => o.name === e.target.value);
                    if (selected) {
                      setOfficeInfo({
                        ...officeInfo,
                        companyName: '太陽シルバーサービス㈱',
                        officeName: selected.name,
                        postal: selected.postal || '',
                        address: selected.address,
                        phone: selected.phone,
                        fax: selected.fax,
                        manager: selected.manager || '',
                      });
                      notify(`${selected.name}を選択しました`);
                    }
                    e.target.value = '';
                  }}
                  defaultValue=""
                  className="w-full p-2 border border-amber-300 rounded outline-none focus:border-amber-500 bg-white text-sm font-medium"
                >
                  <option value="" disabled>▼ 営業所を選んでください</option>
                  {OFFICE_MASTER.map((o) => (
                    <option key={o.name} value={o.name}>
                      {o.name} （〒{o.postal} {o.address.slice(0, 16)}…）
                    </option>
                  ))}
                </select>
                <p className="text-[10px] text-amber-700 mt-1.5">
                  選択すると、営業所名・郵便番号・住所・電話・FAX・所長名が自動入力されます。
                </p>
              </div>

              <p className="text-xs text-slate-600 bg-blue-50 border border-blue-200 rounded p-2.5 mb-2">
                ここで入力した情報が見積書の右上に印字されます。手動でも編集できます。
              </p>
              {[
                { key: 'companyName', label: '会社名 ※', placeholder: '太陽シルバーサービス㈱' },
                { key: 'officeName', label: '営業所名', placeholder: '行橋営業所' },
                { key: 'postal', label: '郵便番号', placeholder: '824-0043' },
                { key: 'address', label: '住所', placeholder: '福岡県行橋市大字流末1327番地' },
                { key: 'phone', label: '電話番号', placeholder: '0930-26-9640' },
                { key: 'fax', label: 'FAX', placeholder: '0930-26-9641' },
                { key: 'email', label: 'Email', placeholder: 'info@example.co.jp' },
                { key: 'manager', label: '所長名', placeholder: '久保 匠史' },
                { key: 'staffName', label: '担当者名', placeholder: '営業　太郎' },
              ].map((field) => (
                <div key={field.key}>
                  <label className="text-xs font-bold text-slate-600 block mb-1">{field.label}</label>
                  <input value={officeInfo[field.key] || ''}
                    onChange={(e) => setOfficeInfo({ ...officeInfo, [field.key]: e.target.value })}
                    placeholder={field.placeholder}
                    className="w-full p-2 border border-slate-300 rounded outline-none focus:border-blue-500" />
                </div>
              ))}
            </div>
            <div className="p-4 border-t flex justify-end gap-2 bg-slate-50 shrink-0">
              <button onClick={() => setShowOfficeModal(false)}
                className="px-5 py-2 bg-slate-200 font-bold rounded hover:bg-slate-300 text-sm">閉じる</button>
              <button onClick={() => { setShowOfficeModal(false); notify('保存しました'); }}
                className="px-5 py-2 bg-amber-500 text-white font-bold rounded hover:bg-amber-600 text-sm flex items-center gap-1">
                <Save className="w-4 h-4" /> 保存して閉じる
              </button>
            </div>
          </div>
        </div>
      )}

      {/* 保存モーダル */}
      {showSaveModal && (
        <div className="fixed inset-0 bg-slate-900/60 z-[80] flex items-center justify-center backdrop-blur-sm p-4 no-print">
          <div className="bg-white p-5 rounded-xl shadow-2xl max-w-md w-full">
            <h3 className="text-lg font-bold text-slate-800 mb-3">見積を保存</h3>
            <input value={saveName} onChange={(e) => setSaveName(e.target.value)}
              placeholder="例: A病院様 2026見積"
              className="w-full border border-slate-300 p-2 rounded mb-3 outline-none focus:border-blue-500 text-sm"
              autoFocus />
            <div className="flex justify-end gap-2">
              <button onClick={() => setShowSaveModal(false)}
                className="px-4 py-2 font-bold text-slate-600 hover:bg-slate-100 rounded text-sm">キャンセル</button>
              <button onClick={handleSave} disabled={!saveName.trim()}
                className={`px-4 py-2 font-bold rounded text-sm text-white ${saveName.trim() ? 'bg-blue-600 hover:bg-blue-700' : 'bg-slate-300 cursor-not-allowed'}`}>
                <Save className="w-4 h-4 inline mr-1" /> 保存
              </button>
            </div>
          </div>
        </div>
      )}

      {/* 保存BOXモーダル */}
      {showLoadModal && (
        <div className="fixed inset-0 bg-slate-900/60 z-[80] flex items-center justify-center backdrop-blur-sm p-4 no-print">
          <div className="bg-white p-5 rounded-xl shadow-2xl max-w-2xl w-full max-h-[80vh] flex flex-col">
            <div className="flex justify-between items-center mb-3 shrink-0">
              <h3 className="text-lg font-bold text-slate-800 flex items-center gap-2">
                <Archive className="w-5 h-5" /> 保存BOX
              </h3>
              <button onClick={() => setShowLoadModal(false)} className="text-slate-400 hover:text-slate-600">
                <X className="w-5 h-5" />
              </button>
            </div>
            <div className="flex-1 overflow-y-auto space-y-2">
              {savedProposals.length === 0 ? (
                <div className="text-center py-12 text-slate-400">
                  <Archive className="w-10 h-10 mx-auto mb-2 text-slate-300" />
                  保存されたデータはありません
                </div>
              ) : savedProposals.map((prop) => (
                <div key={prop.id} className="border border-slate-200 rounded p-3 flex justify-between items-center hover:bg-slate-50">
                  <div className="min-w-0">
                    <div className="font-bold text-slate-800 truncate">{prop.name}</div>
                    <div className="text-[11px] text-slate-500 mt-0.5">
                      {prop.clientName && <span>{prop.clientName}　</span>}
                      {new Date(prop.updatedAt).toLocaleString('ja-JP')}
                    </div>
                  </div>
                  <div className="flex items-center gap-1 shrink-0 ml-2">
                    <button onClick={() => handleLoad(prop)}
                      className="px-3 py-1.5 bg-blue-100 text-blue-700 font-bold rounded text-xs hover:bg-blue-200">開く</button>
                    <button onClick={() => handleDeleteProposal(prop.id)}
                      className="p-1.5 text-slate-400 hover:text-red-600"><Trash2 className="w-4 h-4" /></button>
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}

      {/* 商品マスタモーダル */}
      {showFavoritesModal && (
        <div className="fixed inset-0 bg-slate-900/60 z-[80] flex items-center justify-center backdrop-blur-sm p-4 no-print">
          <div className="bg-white rounded-xl shadow-2xl max-w-2xl w-full max-h-[80vh] flex flex-col">
            <div className="flex justify-between items-center p-4 border-b bg-amber-50 shrink-0">
              <h3 className="text-lg font-bold text-amber-800 flex items-center gap-2">
                <Star className="w-5 h-5 fill-current" /> 商品マスタから追加
              </h3>
              <button onClick={() => setShowFavoritesModal(false)}
                className="text-slate-400 hover:text-slate-600"><X className="w-5 h-5" /></button>
            </div>
            <div className="flex-1 overflow-y-auto p-4 space-y-2">
              {favorites.length === 0 ? (
                <div className="text-center py-10 text-slate-500">
                  <Star className="w-10 h-10 text-slate-300 mx-auto mb-2" />
                  登録された商品はありません<br />
                  <span className="text-xs">商品行右端の★を押すと登録できます</span>
                </div>
              ) : favorites.map((fav) => (
                <div key={fav.id}
                  onClick={() => handleAddFromFavorite(fav)}
                  className="border border-slate-200 rounded p-3 flex justify-between items-center hover:bg-amber-50 cursor-pointer transition">
                  <div className="min-w-0">
                    <div className="font-bold text-slate-800 truncate">{fav.name}</div>
                    <div className="text-[11px] text-slate-500 mt-0.5 flex gap-2">
                      <span>容量: {fav.volume}</span>
                      <span>価格: ¥{Number(fav.price || 0).toLocaleString()}</span>
                    </div>
                  </div>
                  <button className="px-3 py-1 bg-slate-800 text-white font-bold text-xs rounded shrink-0">追加</button>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}

      {/* 使い方ガイドモーダル */}
      {showGuideModal && (
        <div className="fixed inset-0 bg-slate-900/60 z-[80] flex items-center justify-center backdrop-blur-sm p-4 no-print">
          <div className="bg-white rounded-xl shadow-2xl max-w-3xl w-full max-h-[90vh] flex flex-col overflow-hidden">
            <div className="flex justify-between items-center p-4 border-b shrink-0">
              <h3 className="text-lg font-bold text-slate-800 flex items-center gap-2">
                <BookOpen className="w-5 h-5 text-blue-600" /> 使い方ガイド
              </h3>
              <button onClick={() => setShowGuideModal(false)} className="text-slate-400 hover:text-slate-600">
                <X className="w-5 h-5" />
              </button>
            </div>
            <div className="flex-1 overflow-y-auto p-5 space-y-5 bg-slate-50">

              <section className="bg-white p-4 rounded border border-slate-200">
                <div className="flex items-center gap-2 mb-2">
                  <span className="bg-red-500 text-white font-bold w-7 h-7 rounded-full flex items-center justify-center text-sm">
                    <Printer className="w-4 h-4" />
                  </span>
                  <h4 className="font-bold text-base text-slate-800">印刷・PDF出力の手順</h4>
                </div>
                <ol className="list-decimal pl-5 space-y-1.5 text-sm text-slate-600">
                  <li>右上「<strong>印刷/PDF</strong>」ボタンを押します</li>
                  <li>自動的にプレビューモードになり、ブラウザの印刷ダイアログが開きます</li>
                  <li>プリンタを選択して「印刷」、またはプリンタ欄で「<strong>PDFに保存</strong>」を選べばPDFとして保存できます</li>
                  <li>用紙サイズはA4縦に最適化されています</li>
                </ol>
                <div className="mt-2 text-xs text-slate-500 bg-slate-50 p-2 rounded">
                  ヒント: Ctrl+P（Mac は Cmd+P）でも印刷できます。印刷時には原価・操作ボタン・参考資料セクションは自動で非表示になります。
                </div>
              </section>

              <section className="bg-white p-4 rounded border border-slate-200">
                <div className="flex items-center gap-2 mb-2">
                  <span className="bg-amber-500 text-white font-bold w-7 h-7 rounded-full flex items-center justify-center">
                    <Building2 className="w-4 h-4" />
                  </span>
                  <h4 className="font-bold text-base text-slate-800">営業所ごとの設定</h4>
                </div>
                <p className="text-sm text-slate-600 mb-2">
                  「<strong>営業所設定</strong>」ボタンから、会社名・営業所名・住所・電話番号・担当者名などを登録できます。
                  登録内容は自動で保存され、見積書の右上に印字されます。営業所ごとに違う設定を持たせられます。
                </p>
              </section>

              <section className="bg-white p-4 rounded border border-slate-200">
                <div className="flex items-center gap-2 mb-2">
                  <span className="bg-blue-500 text-white font-bold w-7 h-7 rounded-full flex items-center justify-center text-sm">1</span>
                  <h4 className="font-bold text-base text-slate-800">商品の入力</h4>
                </div>
                <p className="text-sm text-slate-600">
                  画面下部の「新しい商品カテゴリを追加」から枠を作成し、商品を登録します。
                  <strong>一番上の行が自動的に「現状品（比較基準）」</strong>となり、2行目以降が提案品として扱われます。
                </p>
              </section>

              <section className="bg-white p-4 rounded border border-slate-200">
                <div className="flex items-center gap-2 mb-2">
                  <span className="bg-blue-500 text-white font-bold w-7 h-7 rounded-full flex items-center justify-center text-sm">2</span>
                  <h4 className="font-bold text-base text-slate-800">モードの使い分け</h4>
                </div>
                <div className="grid grid-cols-1 md:grid-cols-2 gap-2 text-xs">
                  <div className="border border-indigo-100 bg-indigo-50/30 p-2 rounded">
                    <strong className="text-indigo-700">ケース：</strong>「入数」項目が追加され、ケースと単価を自動連動計算
                  </div>
                  <div className="border border-orange-100 bg-orange-50/30 p-2 rounded">
                    <strong className="text-orange-700">原価：</strong>仕入額と目標利益率から販売価格を自動算出（印刷時は非表示）
                  </div>
                  <div className="border border-amber-100 bg-amber-50/30 p-2 rounded">
                    <strong className="text-amber-700">試算：</strong>月使用量から年間コスト比較・削減金額を表示
                  </div>
                  <div className="border border-teal-100 bg-teal-50/30 p-2 rounded">
                    <strong className="text-teal-700">プレビュー：</strong>入力枠を隠し、見積書の最終レイアウトを確認
                  </div>
                </div>
              </section>

            </div>
            <div className="p-3 border-t bg-white flex justify-end shrink-0">
              <button onClick={() => setShowGuideModal(false)}
                className="px-6 py-2 bg-blue-600 text-white font-bold rounded hover:bg-blue-700 text-sm">
                確認しました
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
